# Khai báo thư viện
```
#include<QApplication> // Đây là lớp ứng dụng chính, quản lý vòng lặp sự kiện của chương trình, không có thì không chạy GUI được.

#include<QWidget> // Một QWidget đơn giản có thể chính là một cửa sổ hoặc một container chứa widget khác. (cái để chứa những cái khác)

#include<QPushButton> // lớp tạo nút bấm, tạo các nút tương tác trong phần mềm

#include<QLabel> // lớp để hiển thị văn bản hoặc hình ảnh tĩnh, dùng cho dòng chữ thông báo, hình icon, nhãn của ô nhập liệu, v.v.
```
